CF1303A Erasing Zeroes(水)

https://codeforces.com/contest/1303/problem/A

题目描述

  给定一个由 $01$ 组成的字符串 $s$，你需要去掉一些 000，使得串内所有的 111 都连续。求最小去掉的 000 的个数。$(1\leq |s|\leq 100)$

分析

  找出最左边的 $1$，最右边的 $1$，用两边间隔减去中间 $1$ 个数。

代码

#include<bits/stdc++.h>
using namespace std;
char str[110];
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        cin>>str;
        int len=strlen(str);
        int l=len-1,r=0,ans=0;
        for(int i=0;i<len;i++)
        {
            if(str[i]=='1')
            {
                l=i;
                break;
            }
        }
        for(int i=len-1;i>=0;i--)
        {
            if(str[i]=='1')
            {
                r=i;
                break;
            }
        }
        for(int i=l;i<=r;i++)
            if(str[i]=='1')
                ans++;
        if(ans!=0)
            cout<<r-l+1-ans<<endl;
        else
            cout<<0<<endl;
    }
    return 0;
}

CF1303B National Project(数学)

https://codeforces.com/contest/1303/problem/B

题目描述

  工程队修长度为 $n$ 的路，每天可以修长度为 $1$ 的路或者跳过。每 $g$ 天好天气以后是 $b$ 天坏天气，然后又是 $g$ 天好天气，$b$ 天坏天气，以此类推。在好天气修路的长度至少为总长度的一半，求修完路所需的最小天数（包括跳过的天数）$(1\leq n,g,b\leq 10^9)$。

分析

  先根据间隔算出遇到 $\lceil\frac{n}{2}\rceil$ 天好天气总共需要几天，然后再输出长度和天数的较大值即可（要求路全部修完）。

代码

#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        long long n,g,b,ans;
        cin>>n>>g>>b;
        long long res=(n+1)/2;//需要修的好路长度 
        long long cnt=res/g;//修路的完整周期数 
        if(res%g==0)
            cnt--;
        ans=cnt*(g+b)+(res-cnt*g);
        cout<<max(ans,n)<<endl;
    }
    return 0;
}

CF1303C Perfect Keyboard(dfs)

题目描述

  给出一个字符串 $S$，求是否存在一种 $26$ 个字母排列的构造方式 $T$ 使得 $S$ 中任意相邻两位都在 $T$ 中相邻，数据保证 $S$ 不会存在相邻两位字母相同，若存在则输出 $YES$ 并输出一种合法的构造方式，否则输出 $NO$。$(1\leq |S|\leq 200)$

分析

  每个字母的边集不能大于 $2$，找到 $G[i].size()=1$ 的点进行 dfs，依次将下一个点的边集减一并输出。

代码

#include<bits/stdc++.h>
using namespace std;
vector<int> G[1010];
string str,ans;
int cnt[1010],vis[1010];
void init()
{
    for(int i=0;i<1010;i++)
    {
        cnt[i]=0;
        G[i].clear();
        vis[i]=0;
    }
    ans="";
}
void dfs(int x)
{
    vis[x]=1;
    cnt[x]++;
    ans+='a'+x;
    for(int i=0;i<G[x].size();i++)
        if(!vis[G[x][i]])
            dfs(G[x][i]);
}
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        init();
        bool flag=true;
        map<pair<int,int>,int > mp;
        cin>>str;
        for(int i=1;i<str.size();i++)
        {
            int x=str[i]-'a';
            int y=str[i-1]-'a';
            if(x>y)
                swap(x,y);
            if(mp[make_pair(x,y)]==0)
            {
                G[x].push_back(y);
                G[y].push_back(x);
                mp[make_pair(x,y)]=1;
            }
        }
        for(int i=0;i<26;i++)
        {
            if(G[i].size()>2)
            {
                puts("NO");
                flag=false;
            }
        }
        if(flag)
        {
            for(int i=0;i<26;i++)
                if(!vis[i]&&G[i].size()-cnt[i]<2)
                    dfs(i);
            if(ans.size()==26)
            {
                puts("YES");
                cout<<ans<<endl;
            }
            else
                puts("NO");
        }
    }
    return 0;
}

CF1303D Fill The Bag(二进制+贪心)

题目描述

  有一个大小为 $n$ 的背包和 $m$ 个大小为 $2$ 的非负整数次幂的物品，其中第 $i$ 个物品的大小为 $a_i$。每次可以把一个物品划分成两个相同大小的物品，需要恰好填满整个背包，求最小的划分物品的次数。

$(1\leq n\leq 10^{18},1\leq m\leq 10^5,1\leq a_i\leq 10^9)$

分析

  当 $n>\sum\limits_{i=1}^{m} a_i$ 时，无解。

  贪心策略：

  将 $n$ 二进制拆分为：$n=2^{k_1}+2^{k_2}+···+2^{k_c}$，从低位 $(2^{k_1})$ 到高位 $(2^{k_c})$ 贪心填数。

  假设当前处理到的数字为 $2^{k_i}$。首先考虑位数小于等于 $k_i$ 的所有数中，能不能凑成 $2^{k_i}$ ，若可以，直接填上去；否则考虑分裂位数大于 $k_i$ 的所有数中最小的那一个，使其经过若干次分裂，分裂成 $2^{k_i}$。

代码

#include<bits/stdc++.h>
using namespace std;
int a[100];
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        memset(a,0,sizeof(a));
        long long n,sum=0;
        int m;
        cin>>n>>m;
        for(int i=1;i<=m;i++)
        {
            int x;
            scanf("%d",&x);
            for(int j=0;j<60;j++)
                if((x>>j)&1)
                    a[j]++;
            sum=sum+x;
        }
        if(sum<n)
        {
            puts("-1");
            continue;
        }
        int ans=0;
        for(int i=0;i<60;i++)
        {
            if((n>>i)&1)
            {
                if(a[i]==0)
                {
                    for(int j=i+1;j<70;j++)
                    {
                        if(a[j]!=0)
                        {
                            a[j]--;
                            int now=j;
                            while(now!=i)
                            {
                                now--;
                                a[now]++;
                                ans++;
                            }
                            a[now]++;
                            break;
                        }
                    }
                }
                a[i]--;
            }
            a[i+1]=a[i+1]+a[i]/2;
        }
        cout<<ans<<endl;
    }
    return 0;
}

CF1303E Erase Subsequences(dp)

https://codeforces.com/contest/1303/problem/E

题目描述

  给定两个字符串 $s$ 和 $t$。询问能否找到 $s$ 的两个不相交的子序列（可以为空） $s_1,s_2$ 使 $t=s_1+s_2(1\le|t|\le|s|\le400)$。

分析

  先考虑从 $s$ 种只选取一个子序列操作就能得到 $t$，直接模拟时间复杂度 $O(n)$。

  对于选取两个子序列的情况，先枚举分割点，将 $t$ 分成 $t_1,t_2$。需要从 $s$ 中第一次找出 $t_1$，第二次找出 $t_2$。

  设 $dp[i][j][k]$ 为能否用 $s$ 的前 $i$ 个字符拼出 $t_1$ 的前 $j$ 个字符和 $t_2$ 的前 $k$ 个字符。时间复杂度 $O(n^3)$，总时间复杂度 $O(n^4)$，超时。

  设 $dp[i][j]$ 表示用 $s$ 的前 $i$ 个字符拼出 $t_1$ 的前 $j$ 个字符，$t_2$ 最多能拼出多少个字符。

状态转移方程：$\begin{cases} dp[i+1][j+1]=\max(dp[i+1][j+1],dp[i][j]) & t1[j]==s[i]\ dp[i+1][j]=\max(dp[i+1][j],dp[i][j]+1) & t2[dp[i][j]]==s[i] \ dp[i+1][j]=\max(dp[i+1][j],dp[i][j])\end{cases}$

代码

#include<bits/stdc++.h>
using namespace std;
int dp[500][500];
bool check(string s,string t)
{
    int pos=0;
    if(pos==t.size())
        return true;
    for(int i=0;i<s.size();i++)
    {
        if(s[i]==t[pos])
            pos++;
        if(pos==t.size())
            return true;
    }
    return false;
}
bool _check(string s,string t1,string t2)
{
    memset(dp,-1,sizeof(dp));
    dp[0][0]=0;
    for(int i=0;i<s.size();i++)
    {
        for(int j=0;j<=t1.size();j++)
        {
            if(dp[i][j]>=0)
            {
                if(j<t1.size()&&t1[j]==s[i])
                    dp[i+1][j+1]=max(dp[i+1][j+1],dp[i][j]);
                if(dp[i][j]<t2.size()&&t2[dp[i][j]]==s[i])
                    dp[i+1][j]=max(dp[i+1][j],dp[i][j]+1);
                dp[i+1][j]=max(dp[i+1][j],dp[i][j]);
            }
        }
    }
    int res1=dp[s.size()][t1.size()];
    int res2=t2.size();
    if(res1>=res2)
        return true;
    return false;
}
void solve()
{
    string s,t;
    cin>>s>>t;
    if(check(s,t))
    {
        puts("YES");
        return ;
    }
    for(int i=0;i<t.size()-1;i++)
    {
        string t1="",t2="";
        for(int j=0;j<=i;j++)
            t1+=t[j];
        for(int j=i+1;j<t.size();j++)
            t2+=t[j];
        if(_check(s,t1,t2))
        {
            puts("YES");
            return ;
        }
    }
    puts("NO");
    return ;
}
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        solve();
    }
    return 0;
}
