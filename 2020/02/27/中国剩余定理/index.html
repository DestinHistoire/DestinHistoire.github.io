中国剩余定理

  设 $ m_1,m_2,···,m_n $ 是两两互质的整数，$ m=\displaystyle \prod\limits_{i=1}^{n}m_i,M_i=\frac{m}{m_i} $，$t_i$ 是线性同余方程 $ M_it_i\equiv1(\text{mod}\ m_i)$ 的一个解。对于任意的 $n$ 个整数 $a_1,a_2,···,a_n$，方程组

有整数解，解为 $ x=\displaystyle\sum\limits_{i=1}^{n}a_iM_it_i$。

证明

  因为 $ M_i=\frac{m}{m_i} $ 是除 $m_i$ 之外所有模数的倍数 ，所以 $ \forall k\neq i,a_iM_it_i≡0(\text{mod}\ m_k) $。

  又因为 $ a_iM_it_i≡a_i(\text{mod}\ m_i) $，所以 $ x=\displaystyle \sum\limits_{i=1}^{n}a_iM_it_i $，原方程组成立。

  中国剩余定理给出了模数两两互质的线性同余方程组的一个特解。方程组的通解可以表示为 $ x+km(k\in Z) $。有些题目要求我们求出最小的非负整数解，只需把 $ x $ 对 $ m $ 取模，并让 $ x $ 落在 $ 0 $ ~ $ m-1 $ 的范围即可。

【例题】猜数字 TJOI2009

  现有两组数字，每组 $k$ 个，第一组中的数字分别为：$a_1,a_2,···,a_k$ 表示，第二组中的数字分别用 $b_1,b_2,···,b_k$ 表示。其中第二组中的数字是两两互质的。求最小的非负整数 $n$，满足对于任意的 $i$，$n-a_i$ 能被 $b_i$ 整除。 $ (|a_i|\le 10^9,1\le b_i\le 6\times 10^3,\displaystyle\prod\limits_{i=1}^k b_i\le 10^{18})$。

【分析】

  计算前要将所有 $ a[i]=(a[i]\ \text{mod}\ b[i]+b[i])\ \text{mod}\ b[i]; $

  最后一个点爆 $ \text{long long} $，要用快速乘。

【代码】

#include<bits/stdc++.h>
using namespace std;
int k;
long long a[20],b[20];
long long mul(long long a,long long b,long long mod)
{
    long long ans=0;
    while(b)
    {
        if(b&1) 
            ans=(ans+a)%mod;
        a=(a+a)%mod;
        b>>=1;
    }
    return ans;
}
int exgcd(long long a,long long b,long long &x,long long &y)
{
    if(b==0)
    { 
        x=1; 
        y=0; 
        return a;
    }
    long long gcd=exgcd(b,a%b,x,y);
    int temp=x;x=y;y=temp-a/b*y;
    return gcd;
}
long long CRT()
{
    long long ans=0,lcm=1,x,y;
    for(int i=1;i<=k;i++) 
        lcm=lcm*b[i];
    for(int i=1;i<=k;i++)
    {
        long long temp=lcm/b[i];
        exgcd(temp,b[i],x,y);
        x=(x%b[i]+b[i])%b[i];
        ans=(ans+mul(mul(temp,x,lcm),a[i],lcm))%lcm;
    }
    return (ans+lcm)%lcm;
}
int main()
{
    cin>>k;
    for(int i=1;i<=k;i++) 
        cin>>a[i];
    for(int i=1;i<=k;i++) 
        cin>>b[i];
    for(int i=1;i<=k;i++) 
        a[i]=(a[i]%b[i]+b[i])%b[i];
    cout<<CRT()<<endl;
    return 0;
}

扩展中国剩余定理

  即使模数不满足两两互质，也有办法判断线性同余方程是否有解，并求出方程组的解。

【例题】POJ-2891 Strange Way to Express Integers

  给定 $2n$ 个正整数 $a_1,a_2,\cdots,a_n$ 和 $m_1,m_2,\cdots,m_n$，求一个最小的正整数 $x$ ，满足 $\forall i\in[1,n],x\equiv a_i(\text{mod}\ m_i)$，或者给出无解。

分析

  本题中 $m_i$ 不一定满足两两互质，中国剩余定理不再适用。可以考虑数学归纳法，假设已经求出了前 $k-1$ 个方程构成的方程组的一个解 $x$。记 $m=\displaystyle\prod\limits_{i=1}^{k-1}m_i $，则 $ x+i\ast m(i\in Z) $ 是前 $k-1$ 个方程的通解。

  考虑第 $k$ 个方程，求出一个整数 $t$ ，使得 $ x+t\ast m\equiv a_k(\text{mod}\ m_k)$。该方程等价于 $ m\ast t\equiv a_k-x(\text{mod}\ m_k)$，其中 $t$ 是未知量。这就是一个线性同余方程，可以用扩展欧几里得算法判断是否有解，并求出它的解。若有解，则 $x’=x+t\ast m $ 就是前 $k$ 个方程构成的方程组的一个解。

  综上所述，使用 $n$ 次扩展欧几里得算法，就求出了整个方程组的解。

代码

#include<bits/stdc++.h> 
using namespace std;
int n;
long long A[100010],m[100010]; 
long long exgcd(long long a,long long b,long long &x,long long &y) 
{
    if(b==0) 
    {
        x=1; 
        y=0; 
        return a;
    }
    long long gcd=exgcd(b,a%b,x,y);
    long long  temp=x;x=y;y=temp-a/b*y;
    return gcd;
}
long long exCRT() 
{
    long long x,y;
    long long M=A[1],ans=m[1];
    for(int i=2;i<=n;i++)
    {
        long long a=M,b=A[i],c=(m[i]-ans%A[i]+A[i])%A[i];
        long long gcd=exgcd(a,b,x,y);
        long long d=b/gcd;
        if(c%gcd) 
            return -1;
        x=(x*c/gcd)%d;
        ans=ans+x*M;
        M=M*d;
        ans=(ans%M+M)%M;
    }
    return (ans%M+M)%M;
} 
int main() 
{
    while(cin>>n) 
    {
        for(int i=1;i<=n;i++)
            cin>>A[i]>>m[i];
        cout<<exCRT()<<endl;
    }
    return 0;
}
