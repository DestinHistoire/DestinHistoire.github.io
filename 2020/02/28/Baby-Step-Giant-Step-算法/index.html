Baby-Step-Giant-Step 算法
数论 BSGS算法
算法笔记
发布日期:   2020-02-28

离散对数

  对于给定的 $ a,b,p $ 存在一个 $x$，使得
$$
a^x\equiv b \pmod p
$$
则称 $x$ 为 $b$ 在模 $p$ 意义下以 $a$ 为底的离散对数。

性质

离散对数有类似于一般的对数的性质：

BSGS算法

  需要求解的方程为（$p$ 为质数）
$$
a^x\equiv b\pmod p
$$
  令 $ m=\lceil\sqrt p\rceil $。根据费马小定理，有 $ a^{p-1}\equiv 1\pmod p $，故若方程有解，则必然存在一个 $ 0\leq x<p-1 $。

  设 $ x=im-j $，其中 $ 1\leq i\leq m,0\leq j<m $。

  方程可化为：

  只需要找到一组 $ i、j $ 使得最后一个式子成立即可。

  首先，从 $ 0-m $ 枚举 $j$，递推出右边 $ b\times a^j\ \text{mod}\ p $ 的所有取值，并将其按照 $(b\times a^j\ \text{mod}\ p)\Rightarrow j $ 的映射关系插入到哈希表中。

  之后，从 $ 1-m $ 枚举 $i$，计算 $ (a^m)^i $，每得到一个值后，从哈希表中查找该值，如果存在，取出其对应的 $j$，$ x=im - j $ 即为最小解。

  时间复杂度 $ O(\sqrt p \log p) $。

【例题】P3846 [TJOI2007]可爱的质数

  求最小的 $x$，满足 $ a^x\equiv b(\text{mod}\ p)(2\leq p <2^{31},2\leq a<p,2\leq b<p ) $， $p$ 为质数。

代码

$\text{map}$ 版本

#include<bits/stdc++.h> 
using namespace std;
int p;
map<int,int> mp;
long long quick_pow(long long a,long long b)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
            ans=ans*a%p;
        a=a*a%p;
        b>>=1;
    }
    return ans;
}
int main()
{
    int a,b;
    cin>>p>>a>>b;
    int m=sqrt(p)+1;
    int ans=b;
    for(int i=0;i<m;i++)
    {
        mp[ans]=i;//能更新就更新 
        ans=1ll*ans*a%p;
    }
    int temp=quick_pow(a,m);
    ans=1;
    for(int i=1;i<=m;i++)
    {
        ans=1ll*ans*temp%p;
        if(mp.count(ans))
        {
            printf("%d\n",i*m-mp[ans]);
            return 0;
        }
    }
    puts("no solution");
    return 0;
}

#include<bits/stdc++.h> 
using namespace std;
const int mod=1000007;
int p;
long long quick_pow(long long a,long long b)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
            ans=ans*a%p;
        a=a*a%p;
        b>>=1;
    }
    return ans;
}
struct Hash_Table
{
    int Hash[1000700],value[1000700];
    int _find(int x)
    {
        int t=x%mod;
        while(Hash[t]!=x&&Hash[t]!=-1)
        {
            t=t+107;
            if(t>=mod)
                t=t-mod;
        }
        return t;
    }
    void _insert(int x,int i)
    {
        int t=_find(x);
        Hash[t]=x;
        value[t]=i;
    }
    bool _count(int x)
    {
        int t=_find(x);
        return Hash[t]==x;
    }
    int _value(int x)
    {
        int t=_find(x);
        return value[t];
    }
    void clear()
    {
        memset(value,-1,sizeof(value));
        memset(Hash,-1,sizeof(Hash));
    }
}mp;
int main()
{
    mp.clear();
    int a,b;
    cin>>p>>a>>b;
    int m=sqrt(p)+1;
    int ans=b;
    for(int i=0;i<m;i++)
    {
        mp._insert(ans,i);
        ans=1ll*ans*a%p;
    }
    int temp=quick_pow(a,m);
    ans=1;
    for(int i=1;i<=m;i++)
    {
        ans=1ll*ans*temp%p;
        if(mp._count(ans))
        {
            printf("%d\n",i*m-mp._value(ans));
            return 0;
        }
    }
    puts("no solution");
    return 0;
}

exBSGS

  扩展 $ \text{BSGS} $ 不要求 $p$ 为质数，大致做法和原始的 $ \text{BSGS} $ 一样，只是做些修改。

  令 $ d=\gcd(a,p) $

  若 $ d\nmid b $，那么只有 $y=1$ 时，$x=0$，其他情况均无解。

  若 $ d\mid b$，当 $ d=1 $ 时，直接 $ \text{BSGS} $。

  否则有
$$
a^{x-1}\times \frac{a}{d}\equiv \frac{b}{d}\pmod {\frac{p}{d}}
$$
  继续分解到 $ d=1 $ 为止。
$$
a^{x-t}\times \frac{a^t}{\prod d_i}\equiv \frac{b}{\prod d_i}\pmod {\frac{p}{\prod d_i}}
$$
  然后检验 $ a=[0,t) $ 是否为解，显然 $t$ 是 $\log$ 级别的。

  如果 $ [0,t) $ 都不是解，由于 $x$ 与 $\frac{p}{\prod d_i}$ 互质，$\text{BSGS}$ 求解即可。

  最后的答案需要加上 $t$。

【例题】P4195 【模板】exBSGS/Spoj3105 Mod

  已知数 $a,p,b$，求满足 $ a^x\equiv b(\text{mod}\ p) $ 的最小自然数 $x$（$ a,p,b \leq 10^9$ ，$p$ 不一定是质数）。

代码

#include<bits/stdc++.h>
using namespace std;
const int mod=1000007;
int p;
long long quick_pow(long long a,long long b)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
            ans=ans*a%p;
        a=a*a%p;
        b>>=1;
    }
    return ans;
}
struct Hash_Table
{
    int Hash[1000700],value[1000700];
    int _find(int x)
    {
        int t=x%mod;
        while(Hash[t]!=x&&Hash[t]!=-1)
        {
            t=t+107;
            if(t>=mod)
                t=t-mod;
        }
        return t;
    }
    void _insert(int x,int i)
    {
        int t=_find(x);
        Hash[t]=x;
        value[t]=i;
    }
    bool _count(int x)
    {
        int t=_find(x);
        return Hash[t]==x;
    }
    int _value(int x)
    {
        int t=_find(x);
        return value[t];
    }
    void clear()
    {
        memset(value,-1,sizeof(value));
        memset(Hash,-1,sizeof(Hash));
    }
}mp;
void exBSGS(int a,int b)
{
    if(b==1)
    {
        puts("0");
        return;
    }
    int d=__gcd(a,p),temp=1,t=0;
    while(d^1)
    {
        if(b%d)
        {
            puts("No Solution");
            return;
        }
        t++;
        b=b/d;p=p/d;
        temp=1ll*temp*a/d%p;
        if(b==temp)
        {
            printf("%d\n",t);
            return;
        }
        d=__gcd(a,p);
    }
    int ans=b;
    mp.clear();
    int m=sqrt(p)+1;
    for(int i=0;i<m;i++)
    {
        mp._insert(ans,i);
        ans=1ll*ans*a%p;
    }
    ans=temp;
    temp=quick_pow(a,m);
    for(int i=1;i<=m;i++)
    {
        ans=1ll*ans*temp%p;
        if(mp._count(ans))
        {
            printf("%d\n",i*m-mp._value(ans)+t);
            return;
        }
    }
    puts("No Solution");
}
int main()
{
    int a,b;
    while(~scanf("%d%d%d",&a,&p,&b)&&(a||p||b))
    {
        a=a%p;b=b%p;
        exBSGS(a,b);
    }
    return 0;
}
