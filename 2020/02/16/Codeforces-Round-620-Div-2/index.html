CF1304A Two Rabbits(水)

https://codeforces.com/contest/1304/problem/A

题目描述

  一只兔子位于 $x$ 坐标，每秒跳 $a$ 个单位，另一只位于 $y$ 坐标，每秒跳 $b$ 个单位，保证 $x<y$，两只兔子相向而行。求两只兔子能否在整数秒相遇，如果能，输出相遇的那一秒，如果无法在整数秒相遇，输出 $-1$ $(0 \le x \lt y \le 10^9 ,1 \le a,b \le 10^9)$。

代码

#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        long long  x,y,a,b;
        cin>>x>>y>>a>>b;
        if((y-x)%(a+b)==0)
            cout<<(y-x)/(a+b)<<endl;
        else
            cout<<"-1"<<endl;
    }
    return 0;
}

CF1304B Longest Palindrome(暴力)

https://codeforces.com/contest/1304/problem/B

题目描述

  给定 $n$ 个长度为 $m$ 的字符串，所有字符串互不相同。删去其中的几个（或者不删），使得剩下字符串经过排列后所形成的回文串长度最大。输出最大的长度和那个回文串。$(1\leq n\leq 100,1\leq m\leq 50)$

分析

  显然 “删去若干个字符串” 相当于 “选出若干个字符串”。

  因为所有字符串的长度均相同，所以在最后拼接成的回文串中，选出的第一个串与最后一个串一定是恰好 “相反” 的，第二个串与倒数第二个串也是恰好 “相反” 的。

  假设最后依次选出了 $t_1, t_2, …, t_k$ 这些字符串，那么 $t_i=\text{reverse}(t_{k-i+1})$。

  所以对于每个字符串，判断它的翻转是否在 $n$ 个字符串中。如果在，说明这一对 “相反” 的字符串可以分别添加到答案的头和尾。如果它的翻转不在 $n$ 个字符串中，那么它就不可能添加到答案中。

  接着可以发现，答案里 正中间 的一个字符串（如果存在，即选择了奇数个字符串）一定是回文的。那么只要找到任意一个回文的字符串，将它放在答案的中间即可。如果没有回文串，那么就没有一个字符串可以成为答案的 “正中间”，即我们只能选出偶数个字符串（成对）。

  所以最后答案就是 若干对互为“翻转”的字符串（在头尾） + 中间的一个回文串（如果存在）。

代码

#include<bits/stdc++.h>
using namespace std;
string str[110];
bool vis[110];
bool check(string A,string B)
{
    reverse(A.begin(),A.end());
    return A==B;
}
int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++)
        cin>>str[i];
    string ans;
    for(int i=1;i<=n;i++)
    {
        for(int j=i+1;j<=n;j++)
        {
            if(vis[i]==true||vis[j]==true)
                continue;
            bool flag=true;
            if(!check(str[i],str[j]))
                flag=false;
            if(flag)
            {
                vis[i]=true;
                vis[j]=true;
                ans=ans+str[i];
            }
        }
    } 
    string _ans="";
    for(int i=1;i<=n;i++)
    {
        if(vis[i]==0)
        {
            bool flag=true;
            for(int j=0;j<m;j++)
                if(str[i][j]!=str[i][m-j-1])
                    flag=false;
            if(flag==true)
            {
                _ans=str[i];
                break;
            }
        }
    }
    cout<<2*ans.size()+_ans.size()<<endl;
    cout<<ans;
    reverse(ans.begin(),ans.end());
    cout<<_ans;
    cout<<ans<<endl;
    return 0;
}

CF1304C Air Conditioner(思维)

https://codeforces.com/contest/1304/problem/C

题目描述

  餐馆中有一个空调，每分钟可以选择上调 $1$ 个单位的温度或下调 $1$ 个单位的温度，也可以选择不变，初始的温度为 $m$ 。

  有 $n$ 个食客，每个食客会在 $t_i$ 时间点到达，他所能适应的最低温度是 $l_i$，最高温度是 $h_i$，他只会在 $t_i$ 时刻逗留。

  如果温度不在食客的适应范围内，他就会不舒服，判断空调能否使得 $n$ 位来就餐的食客都感到舒服。

$(1\le n\le 100,-10^9\le m,l_i,h_i\le 10^9,1\le t_i\le 10^9)$。

分析

  维护每个时刻，温度可能取到的区间 $[L,R]$ 。初始状态下，$t=0,[L,R]=[m,m]$。

  在某个时刻，温度区间是 $[L,R]$ ，过了 $d$ 时间，温度区间将会变成 $[L-d,R+d]$。

  按照顾客到达的时间顺序来处理，由于要满足第 $i$ 个顾客的需求，将 $[L-d,R+d]$ 与 $[l_i,r_i]$ 取交集即可，交集：$[L,R]\bigcap[l_i,r_i]=[\max(L,l_i),\max(R,r_i)]$。如果此时集合为空，说明不可能。

  时间复杂度：$O(n)$。

代码

#include<bits/stdc++.h>
using namespace std;
struct node
{
    int l,r,t;
}a[110];
bool cmp(node A,node B)
{
    return A.t<B.t;
}
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        int n,m,l,r;
        cin>>n>>m;
        l=m,r=m;
        int t=0;
        bool flag=true;
        for(int i=1;i<=n;i++)
            cin>>a[i].t>>a[i].l>>a[i].r;
        sort(a+1,a+1+n,cmp);
        for(int i=1;i<=n;i++)
        {
            int temp=a[i].t-t;
            l=l-temp;r=r+temp;
            l=max(l,a[i].l);r=min(r,a[i].r);
            if((l<=r&&l<=a[i].r&&a[i].l<=r)==false)
                flag=false;
            t=a[i].t;
        }
        if(flag)
            puts("YES");
        else
            puts("NO");
    }
    return 0;
}

CF1304D Shortest and Longest LIS(构造)

https://codeforces.com/contest/1304/problem/D

题目描述

  有一个 $1$ 到 $n$ 的排列，知道所有相邻两个位置的大小关系，构造出两个排列，满足第一个排列的 LIS 尽可能短，第二个排列的 LIS 尽可能长。。$(1\leq n\leq 2·10^5)$

分析

  构造 LIS 最短，统计连续的 $<$，把大的数字尽量放到前面；构造 LIS 最长，统计连续的 $>$，把小的数字尽量放到前面。

代码

#include<bits/stdc++.h>
using namespace std;
char str[200010];
int a[200010],b[200010],n,m;
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        cin>>n;
        scanf("%s",str+2);
        int temp=n,cnt=0;
        for(int i=2;i<=n;i++)
        {
            if(str[i]=='<')
                cnt++;
            if(str[i]=='>')
            {
                for(int i=temp-cnt;i<=temp;i++)
                    printf("%d ",i);
                temp=temp-cnt-1;
                cnt=0;
            }
        }
        for(int i=temp-cnt;i<=temp&&i>=1;i++)
            printf("%d ",i);
        cout<<endl;
        temp=1;cnt=0;
        for(int i=2;i<=n;i++)
        {
            if(str[i]=='>')
                cnt++;
            if(str[i]=='<')
            {
                for(int i=temp+cnt;i>=temp;i--)
                    printf("%d ",i);
                temp=temp+cnt+1;
                cnt=0;
            }
        }
        for(int i=temp+cnt;temp<=i&&i<=n;i--)
            printf("%d ",i);
        cout<<endl;
    }
    return 0;
}

CF1304E 1-Trees and Queries(LCA)

https://codeforces.com/contest/1304/problem/E

题目描述

  给定一个 $n$ 个点的树，相邻点的距离为 $1$。

  $q$ 个询问，每个询问包含 $5$ 个整数：$x,y,a,b,k$，含义是在原树上新连上一条边 $(x,y)$，判断从 $a$ 点到 $b$ 点是否有距离为 $k$ 的路径。

注意：

  $1.$每个询问是独立的，即上次询问加上的边，不能为这一次的询问所用。

  $2.$这一条路径也许会重复经过某一条边或某一点。

  $(3\le n\le 10^5,1\le q\le 10^5,u,v,a,b,x,y\in[1,n],u\ne v,x\ne y,k\in [1,10^9])$

分析

  可能的简单路径只有三种：

  $1.a\rightarrow b$

  $2.a\rightarrow x\rightarrow y\rightarrow b$

  $3.a\rightarrow y\rightarrow x \rightarrow b$

  只要有 $1$ 条路径满足一下两个条件，就说明存在这样一条长度为 $k$ 的路径。

  $1.$路径长度 $\leq k $

  $2.$路径长度和 $k$ 奇偶性相同

  奇偶性相同，代表两个数的差是偶数，可以在某条边上来回走。

代码

#include<bits/stdc++.h>
using namespace std;
const int SIZE=100010;
int head[SIZE],f[SIZE][20],dep[SIZE];
int n,m,num_edge,t;
queue<int> Q;
struct Edge
{
    int to;
    int Next;
}edge[SIZE<<1];
void add_edge(int from,int to)
{
    edge[++num_edge].to=to;
    edge[num_edge].Next=head[from];
    head[from]=num_edge;
}
void bfs()
{
    dep[1]=1;
    Q.push(1);
    while(!Q.empty())
    {
        int x=Q.front();
        Q.pop();
        for(int i=head[x];i;i=edge[i].Next)
        {
            int y=edge[i].to;
            if(dep[y])
                continue;
            dep[y]=dep[x]+1;
            f[y][0]=x;
            for(int j=1;j<=t;j++)
                f[y][j]=f[f[y][j-1]][j-1];
            Q.push(y);
        }
    }
}
int lca(int x,int y)
{
    if(dep[x]>dep[y])
        swap(x,y);
    for(int i=t;i>=0;i--)
        if(dep[f[y][i]]>=dep[x])
            y=f[y][i];
    if(x==y)
        return x;
    for(int i=t;i>=0;i--)
        if(f[x][i]!=f[y][i])
            x=f[x][i],y=f[y][i];
    return f[x][0];
}
int dis(int x,int y)
{
    return dep[x]+dep[y]-2*dep[lca(x,y)];
}
bool check(int x,int y)
{
    return x<=y&&x%2==y%2;
}
int main()
{
    cin>>n;
    t=(int)(log(n)/log(2))+1;
    for(int i=1;i<=n-1;i++)
    {
        int u,v;
        scanf("%d%d",&u,&v); 
        add_edge(u,v);
        add_edge(v,u);
    }
    bfs();
    cin>>m;
    while(m--)
    {
        int x,y,a,b,k;
        scanf("%d%d%d%d%d",&x,&y,&a,&b,&k);
        if(check(dis(a,b),k)||check(dis(a,x)+dis(y,b)+1,k)||check(dis(a,y)+dis(x,b)+1,k))
            puts("YES");
        else
            puts("NO");
    }
    return 0;
}
