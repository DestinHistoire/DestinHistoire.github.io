A.细胞分裂(质因数分解)

https://ac.nowcoder.com/acm/contest/948/A

题目描述

  有 $n$ 种细胞，编号从 $1$ ~ $n$，一个第 $i$ 种细胞经过 $1$ 秒钟可以分裂为 $S_i$ 个同种细胞（ $ S_i $为正整数），需要的细胞总数为 $ M={m_1}^{m_2} $，整个实验过程中不允许分割单个细胞。

  求选择哪种细胞培养，可以使得实验的开始时间最早 $ (1\leq n\leq 10000,1\leq m_1\leq 30000,1\leq m_2\leq 10000,1\leq S_i\leq 2\times 10^9)$。

分析

  将 $m_1$ 唯一分解，然后遍历每个细胞，将每个细胞对应的每个质因子所需要的因子数目记录下来，求每个最大值，再求每个细胞最大值的最小值。

代码

#include<bits/stdc++.h>
using namespace std;
int p[100010],c[100010];
int main()
{
    int n,m1,m2;
    cin>>n>>m1>>m2;
    int cnt=0;
    for(int i=2;i*i<=m1;i++)
    {
        if(m1%i==0)
        {
            p[++cnt]=i;
            while(m1%i==0)
            {
                m1=m1/i;
                c[cnt]++;
            }
        }
    }
    if(m1!=1)
        p[++cnt]=m1,c[cnt]++;
    int ans=0x3f3f3f3f;
    while(n--)
    {
        bool flag=true;
        int x;
        scanf("%d",&x);
        int temp=0;
        for(int i=1;i<=cnt;i++)
        {
            if(x%p[i]==0)
            {
                int num=0;
                while(x%p[i]==0)
                {
                    x=x/p[i];
                    num++;
                }
                if(c[i]*m2%num==0)
                    temp=max(temp,c[i]*m2/num);
                else
                    temp=max(temp,c[i]*m2/num+1);
            }
            else
            {
                flag=false;
                break;
            }
        }
        if(flag)
            ans=min(ans,temp);
    }
    if(ans==0x3f3f3f3f)
        puts("-1");
    else
        cout<<ans<<endl;
    return 0;
}

B.A题(思维)

https://ac.nowcoder.com/acm/contest/948/B

题目描述

  A要去B的城市游玩，A在城市 $1$ 居住，B在城市 $x$ 居住，现在有一些神奇的传送门和一些奇神的传送门。
  已知，神奇的传送门可以从编号小的城市传送往编号大的城市，奇神的传送门可以从编号大的城市传送往编号小的城市，但是在某些城市没有某种传送门。
  已知数组 $a$，其中 $a_i$ 代表着城市i是否有神奇的传送门（$ a_i=1$ 代表有，$a_i=0$ 代表没有），以及数组 $b$，其中 $ b_i $ 代表着城市 $i$ 是否有奇神的传送门。

  判断A能不能到B的城市。

（$2 \leq n\leq 1000,2\leq x\leq 1000$）

分析

  首先判断特殊情况：若 $ a[1]=0 $，则城市 $1$ 无法到达任意城市；若 $a[x]=b[x]=0$，则无法到达城市 $x$；若 $a[x]=1$，则从城市 $1$ 出发一定可以到达城市 $x$。

  对于其余情况，必须经过其他城市才能到达 $x$，遍历 $x$ 之后的城市，判断是否可以双向到达。

代码

#include<bits/stdc++.h>
using namespace std;
int a[1010],b[1010];
int main()
{
    int n,x;
    while(cin>>n>>x)
    {
        for(int i=1;i<=n;i++)
            scanf("%d",&a[i]);
        for(int i=1;i<=n;i++)
            scanf("%d",&b[i]);
        if(a[1]==0)
            puts("NO");
        else if(a[x]==0&&b[x]==0)
            puts("NO");
        else if(a[x]==1)
            puts("YES");
        else
        {
            bool flag=false;
            for(int i=x+1;i<=n;i++)
                if(a[i]==1&&b[i]==1)
                    flag=true;
            if(flag)
                puts("YES");
            else
                puts("NO");
        }
    }
    return 0;
}

C.均分糖果(贪心)

https://ac.nowcoder.com/acm/contest/948/C

题目描述

NOIP2002 均分纸牌

  有 $n$ 堆糖果，编号分别为 $ 1,2,···,n $。每堆上有若干个，但糖果总数必为 $n$ 的倍数。可以在任一堆上取若干个糖果，然后移动。

  移动规则为：在编号为 $1$ 的堆上取的糖果，只能移到编号为 $2$ 的堆上；在编号为 $n$ 的堆上取的糖果，只能移到编号为 $ n-1 $ 的堆上；其他堆上取的糖果，可以移到相邻左边或右边的堆上。

  现在要求找出一种移动方法，用最少的移动次数使每堆上糖果数都一样多。$(1\leq n\leq 100,1\leq a_i\leq 10000)$

分析

  算每堆糖果平均数，求每堆纸牌与平均数的关系，当 $ a[i] $（第 $x$ 堆纸牌与平均数的关系）不等于 $0$ 时，$a[i+1]=a[i+1]+a[i]-sum $，移动次数加 $1$。

代码

#include<bits/stdc++.h>
using namespace std;
int n,a[1010];
int main()
{
    int n;
    while(cin>>n)
    {
        int sum=0;
        for(int i=1;i<=n;i++)
        {
            scanf("%d",&a[i]);
            sum=sum+a[i];
        }
        sum=sum/n;
        int cnt=0;
        for(int i=1;i<=n;i++)
            if(a[i]-sum!=0)
                a[i+1]=a[i+1]+a[i]-sum,cnt++;
        cout<<cnt<<endl;
    }
    return 0;
}

D.B题(dfs)

https://ac.nowcoder.com/acm/contest/948/D

题目描述

  有一个连通图，包含 $n$ 个点 $n$ 条无向边，其中每个点都与其他的两个点直接相连 (即这是一个环)，现在这个环的边变成了有向边，变成了有向边后得到的有向图不一定是强连通的（强连通图是指一个有向图中任意两点$v_1、v_2$ 间存在 $v_1$ 到 $v_2$ 的路径及 $v_2$ 到 $ v_1 $ 的路径的图）。

  现在给出 $n$ 条有向边和把某条有向边转换方向后的代价，问要使输入的有向图变成一个强连通图，花费代价最小是多少$(3\leq n\leq 100,1\leq u_i,v_i\leq n,1\leq w_i\leq 100)$。

分析

  把有向图建为特殊的无向图，正向时边的权值为 $0$，反向权值为原来的权值，再搜索图的时候，没改动的边花费为 $0$，改动的边花费对应的权值。

  从点 $1$ 开始顺时针和逆时针搜索，记录最小花费即可。

代码

#include<bits/stdc++.h>
using namespace std;
int head[1010],num_edge;
int n,ans=0x3f3f3f3f;
struct Edge
{
    int to;
    int dis;
    int Next;
}edge[1010*2];
void add_edge(int from,int to,int dis)
{
    edge[++num_edge].to=to;
    edge[num_edge].dis=dis;
    edge[num_edge].Next=head[from];
    head[from]=num_edge;
}
void dfs(int x,int fa,int sum)
{
    if(x==1)
    {
        ans=min(ans,sum);
        return ;
    }
    else if(sum>ans)
        return ;
    for(int i=head[x];i;i=edge[i].Next)
    {
        if(fa==edge[i].to)
            continue;
        dfs(edge[i].to,x,sum+edge[i].dis);
    }
}
int main()
{
    while(cin>>n)
    {
        memset(head,0,sizeof(head));
        num_edge=0;
        for(int i=1;i<=n;i++)
        {
            int u,v,w;
            scanf("%d%d%d",&u,&v,&w);
            add_edge(u,v,0);
            add_edge(v,u,w);
        }
        ans=0x3f3f3f3f;
        for(int i=head[1];i;i=edge[i].Next)
        {
            int v=edge[i].to;
            int w=edge[i].dis;
            dfs(v,1,w);
        }
        cout<<ans<<endl;
    }
    return 0;
}

题目描述

NOIP2010 数字统计

  统计 $[l,r]$ 中 $2$ 出现的次数$(1\leq l\leq r\leq 10000)$。

暴力

#include<bits/stdc++.h>
using namespace std;
int main()
{
    int l,r;
    while(cin>>l>>r)
    {
        int ans=0;
        for(int i=l;i<=r;i++)
            for(int j=i;j!=0;j=j/10)
                if(j%10==2)
                    ans++;
        cout<<ans<<endl;
    }
    return 0;
}

数位dp

#include<bits/stdc++.h>
using namespace std;
int a[20];
long long dp[20][20];
int dgt=2;
long long dfs(long long pos,long long st,int lead,int limit)
{
    if(pos==0)
        return st;
    if(dp[pos][st]!=-1&&limit==0&&lead==0)
        return dp[pos][st];
    long long ans=0;
    long long res=limit?a[pos]:9;
    for(int i=0;i<=res;i++)
        ans=ans+dfs(pos-1,st+(i==2),lead&&i==0,limit&&i==res);
    if(limit==0&&lead==0)
        dp[pos][st]=ans;
    return ans;
}
long long solve(long long x)
{
    int cnt=0;
    while(x)
    {
        a[++cnt]=x%10;
        x=x/10;
    }
    memset(dp,-1,sizeof(dp));
    return dfs(cnt,0,1,1);
}
int main()
{
    int l,r;
    while(cin>>l>>r)
    {
        cout<<solve(r)-solve(l-1)<<endl; 
    }
    return 0;
}

F.最大公约数和最小公倍数问题(数学)

https://ac.nowcoder.com/acm/contest/948/F

题目描述

NOIP2001 最大公约数和最小公倍数问题

  输入 $2$ 个正整数 $x_0,y_0(2\leq x_0\leq 10^5,2\leq y_0\leq 10^5)$，求出满足下列条件的 $P,Q$ 的个数。

条件：$ 1.P,Q $ 是正整数；$2.$ 要求 $ P,Q $ 以 $ x_0 $ 为最大公约数，以 $ y_0 $ 为最小公倍数。

代码

#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long x,y;
    while(cin>>x>>y)
    {
        int ans=0;
        for(long long i=2;i*i<=x*y;i++)
            if(x*y%i==0&&__gcd(i,x*y/i)==x)
                ans++;
        ans=ans*2;
        cout<<ans<<endl;
    }
    return 0;
}

G.毕业生的纪念礼物(优先队列+贪心)

https://ac.nowcoder.com/acm/contest/948/G

题目描述

  有n个纪念品，每个纪念品都有一个种类 $ a[ i ] $，现在要求对每个毕业生分发三个种类不同的纪念品，现在需要你来计算下共可以发给多少个毕业生？$(1\leq n\leq 10^5,1\leq a[i]\leq 10^9)$

分析

 用 $ \text{map} $ 将所有的种类保存下来，再用优先队列记录，每次弹出三个数量最多的礼物并更新次数即可。

代码

#include<bits/stdc++.h>
using namespace std;
map<int,int> mp;
priority_queue<int,vector<int>,less<int> > Q;//大根堆
int main()
{
    int n,ans=0;
    cin>>n;
    for(int i=1;i<=n;i++)
    {    
        int x;
        scanf("%d",&x);
        mp[x]++;
    }
    for(auto it:mp)
        Q.push(it.second);
    while(Q.size()>=3)
    {
        int a=Q.top();
        Q.pop();
        int b=Q.top();
        Q.pop();
        int c=Q.top();
        Q.pop();
        a--,b--,c--;
        if(a!=0)Q.push(a);
        if(b!=0)Q.push(b);
        if(c!=0)Q.push(c);
        ans++;
    }
    cout<<ans<<endl;
    return 0;
}

H.毕业生的序列游戏(概率dp)

https://ac.nowcoder.com/acm/contest/948/H

题目描述

codeforces908D New Year and Arbitrary Arrangement

  给定三个数 $ k,p_a,p_b $，每次有 $ \frac{p_a}{p_a+p_b} $ 的概率往后面添加一个 $’a’$，有 $ \frac{p_b}{p_a+p_b} $ 的概率往后面添加一个 $’b’$，当出现了 $k$ 个形如 $’ab’$ 的子序列（不用连续）时停止。求最后的形如 $’ab’$ 的子序列个数的期望。答案对 $1e9+7$ 取模$ (1\leq k\leq 1000,1\leq p_a,p_b\leq 1000000) $。

分析

  设 $\text{dp}[i][j]$ 为该序列前缀为 $i$ 个 $a$，有 $j$ 个子序列 $ ‘ab’ $ 时，出现子序列 $ ‘ab’ $ 的次数的期望。

  状态转移方程：
$$
\text{dp}[i][j]=\text{dp}[i+1][j]\frac{p_a}{p_a+p_b}+\text{dp}[i][i+j]\frac{p_b}{p_a+p_b}
$$
  如果出现无穷多 $a$ 就是不出现 $b$ ，这种情况就会无限 $ \text{dp} $ 下去。

  如果 $i+j\geq k$，只需要再出现一个 $b$，就能满足要求，需要手算一个无穷级数：

  错位相减得：

代码

#include<bits/stdc++.h>
using namespace std;
const int mod=1e9+7;
long long k,pa,pb,invb,invab;
long long dp[1010][1010];
long long quick_pow(long long a,long long b)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
            ans=ans*a%mod;
        a=a*a%mod;
        b>>=1;
    }
    return ans;
}
long long dfs(long long x,long long y)
{
    if(dp[x][y]!=0)
        return dp[x][y];
    if(x+y>=k)
        return dp[x][y]=(x+y+pa*invb%mod)%mod;
    return dp[x][y]=(pa*invab%mod*dfs(x+1,y)%mod+pb*invab%mod*dfs(x,x+y)%mod)%mod;
}
int main()
{
    cin>>k>>pa>>pb;
    invb=quick_pow(pb,mod-2);
    invab=quick_pow(pa+pb,mod-2);
    cout<<dfs(1,0)<<endl;
    return 0;
}

J.你的粪坑v2(四维dp)

题目描述

  起点为 $ (1,1) $ ，终点为 $(n,n)$， 每个点都有一个权值，只能向下或向右走，并且每次换方向时都会附加一些权值，大小为 $ 2^{k-1} $（$ k $ 为转换次数），输出最小权值和$(1\leq n\leq 100,1\leq a_{ij}\leq 100)$。

分析

  设 $\text{dp}[i][j][k][0/1]$ 表示当前在点 $(i,j)$ ，已经换了 $k$ 次方向，$0$ 表示当前方向向右，$1$ 表示当前方向向下的最小权值。

状态转移方程：

代码

#include<bits/stdc++.h>
using namespace std;
int a[110][110];
int dp[110][110][210][3];
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        memset(dp,0x3f3f3f3f,sizeof(dp));
        int n;
        cin>>n;
        for(int i=1;i<=n;i++)
            for(int j=1;j<=n;j++)
                cin>>a[i][j];
        dp[1][1][0][0]=dp[1][1][0][1]=a[1][1];
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=n;j++)
            {
                for(int k=0;k<=min(n,20);k++)//换方向次数不能太多
                {
                    dp[i][j][k][0]=min(dp[i][j][k][0],dp[i][j-1][k][0]+a[i][j]);
                    if(k!=0)
                        dp[i][j][k][0]=min(dp[i][j][k][0],dp[i-1][j][k-1][1]+a[i][j]+(1<<(k-1)));
                    dp[i][j][k][1]=min(dp[i][j][k][1],dp[i-1][j][k][1]+a[i][j]);
                    if(k!=0)
                        dp[i][j][k][1]=min(dp[i][j][k][1],dp[i][j-1][k-1][0]+a[i][j]+(1<<(k-1)));
                }
            }
        }
        int ans=0x3f3f3f3f;//由于无法确定换了多少次方向，所以要遍历换方向次数求最小
        for(int k=0;k<=min(n,20);k++)
            ans=min(ans,min(dp[n][n][k][0],dp[n][n][k][1]));
        cout<<ans<<endl;
    }
    return 0;
}
