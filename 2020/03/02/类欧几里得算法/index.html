类欧几里得算法
数论 类欧几里得算法
算法笔记
发布日期:   2020-03-02

  类欧几里得算法是统计直线下整点贡献的一种算法，最常见的类欧几里得算法是求：
$$
\displaystyle\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$
也就是直线 $ \frac{ax+b}{c} $ 下纵坐标大于 $0$ 的整点数。

常见形式

  给定 $a,b,c,n$，求：
$$
f(a,b,c,n)=\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$

$$
g(a,b,c,n)=\sum\limits_{i=0}^{n}i\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$

$$
h(a,b,c,n)=\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor^2
$$

前置数学关系

$$
\Big\lfloor\frac{a}{b} \Big\rfloor\geq c \Longleftrightarrow a\geq bc
$$

$$
\Big\lceil\frac{a}{b} \Big\rceil\leq c \Longleftrightarrow a\leq bc
$$

$$
\Big\lfloor\frac{a}{b} \Big\rfloor< c \Longleftrightarrow a< bc
$$

$$
\Big\lceil\frac{a}{b} \Big\rceil> c \Longleftrightarrow a> bc
$$

  如果有等号的话就说明可以挨到，那就让他尽量接近，所以取整取离 $c$ 近的那一边。反之，如果没有等号说明挨不到，那么就要让它尽量避免“接触” $c$，所以取整取离 $c$ 远的那一边。
$$
\Big\lceil\frac{a}{b} \Big\rceil=\Big\lfloor \frac{a+b-1}{b} \Big\rfloor
$$

$$
\Big\lfloor\frac{a}{b} \Big\rfloor=\Big\lceil \frac{a+b+1}{b} \Big\rceil
$$

  因为 $“−1/+1”$ 相当于把取整符号”移到下面/上面”，但是因为取整本身差 $“1”$ ，于是就要相应地$+b/−b$。

推导

$f(a,b,c,n)$

$$
f(a,b,c,n)=\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$

  当 $a\geq c$ 或 $b\geq c$ 时，

  当 $a<c$ 且 $b<c$ 时，

  可以发现右边递归成为子问题，继续一轮操作即可。边界条件：$a=0$。

$g(a,b,c,n)$

$$
g(a,b,c,n)=\sum\limits_{i=0}^{n}i\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$

  当 $a\geq c$ 或 $b\geq c$ 时，

  当 $a<c$ 且 $b<c$ 时，

  可以发现需要继续推导 $h(a,b,c,n)$。

$h(a,b,c,n)$

$$
h(a,b,c,n)=\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor^2
$$

  当 $a\geq c$ 或 $b\geq c$ 时，

  当 $a<c$ 且 $b<c$ 时，

  由于 $n^2=2\times \frac{n(n+1)}{2}-n=2\displaystyle\sum\limits_{i=1}^{n}i-n$

  可以得到：

例题

P5170 【模板】类欧几里得算法

  给定 $n,a,b,c(0\leq n,a,b,c\leq 10^9,c\neq 0)$，分别求：
$$
f(a,b,c,n)=\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$

$$
h(a,b,c,n)=\sum\limits_{i=0}^{n}\left\lfloor\dfrac{ai+b}{c}\right \rfloor^2
$$

$$
g(a,b,c,n)=\sum\limits_{i=0}^{n}i\left\lfloor\dfrac{ai+b}{c}\right \rfloor
$$

代码

#include<bits/stdc++.h>
using namespace std;
const long long mod=998244353;
const long long inv2=499122177;
const long long inv6=166374059;
struct Node
{
    long long f,g,h;
};
Node solve(long long a,long long b,long long c,long long n)
{
    Node ans,temp;
    if(a==0)
    {
        ans.f=(n+1)*(b/c)%mod;
        ans.g=(b/c)*n%mod*(n+1)%mod*inv2%mod;
        ans.h=(n+1)*(b/c)%mod*(b/c)%mod;
        return ans;
    }
    if(a>=c||b>=c)
    {
        temp=solve(a%c,b%c,c,n);
        ans.f=(temp.f+(a/c)*n%mod*(n+1)%mod*inv2%mod+(b/c)*(n+1)%mod)%mod;
        ans.g=(temp.g+(a/c)*n%mod*(n+1)%mod*(2*n+1)%mod*inv6%mod+(b/c)*n%mod*(n+1)%mod*inv2%mod)%mod;
        ans.h=((a/c)*(a/c)%mod*n%mod*(n+1)%mod*(2*n+1)%mod*inv6%mod + 
                (b/c)*(b/c)%mod*(n+1)%mod+(a/c)*(b/c)%mod*n%mod*(n+1)%mod+ 
                temp.h+2*(a/c)%mod*temp.g%mod+2*(b/c)%mod*temp.f%mod)%mod;
        return ans;        
    }
    long long m=(a*n+b)/c;
    temp=solve(c,c-b-1,a,m-1);
    ans.f=(n*(m%mod)%mod-temp.f)%mod;
    ans.g=(n*(n+1)%mod*(m%mod)%mod-temp.f-temp.h)%mod*inv2%mod;
    ans.h=(n*(m%mod)%mod*((m+1)%mod)%mod-2*temp.g-2*temp.f-ans.f)%mod;
    return ans;
}
int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        long long n,a,b,c;
        scanf("%lld%lld%lld%lld",&n,&a,&b,&c);
        Node ans=solve(a,b,c,n);
        printf("%lld %lld %lld\n",(ans.f+mod)%mod,(ans.h+mod)%mod,(ans.g+mod)%mod);
    }
    return 0;
}
